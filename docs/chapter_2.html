Using ggplot

(plot->svg
 (let [x (r->clj (r "1: 7"))
       y (r->clj (r "c(77, 72, 75, 73, 66, 64, 59)"))]
   (-> {:x x :y y}
       dataset/->dataset
       (ggplot (aes :x x
                    :y y))
       (r/r+ (geom_point)
             (xlab "x")
             (ylab "y")))))
