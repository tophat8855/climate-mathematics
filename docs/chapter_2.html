2.2.2 Using ggplot

(plot->svg
 (let [x (r->clj (r "1: 7"))
       y (r->clj (r "c(77, 72, 75, 73, 66, 64, 59)"))]
   (-> {:x x :y y}
       dataset/->dataset
       (ggplot (aes :x x
                    :y y))
       (r/r+ (geom_point)
             (xlab "x")
             (ylab "y")))))
(plot->svg
 (let [{:keys [x y] :as sin-data} (r->clj (r "plot(sin, -pi, 2*pi)"))]
   (-> sin-data
       dataset/->dataset
       sin-data
       (ggplot (aes :x x
                    :y y))
       (r/r+ (geom_path)
             (xlab "x")
             (ylab "y")))))
(r "square <- function(x) x*x")
function (x) 
x * x

(def x (r->clj (r "x <- seq(-3, 2, length=100)")))
(def y (r->clj (r "y <- square(x)")))
(plot->svg
 (-> {:x (r->clj x) :y (r->clj y)}
     dataset/->dataset
     (ggplot (aes :x x
                  :y y))
     (r/r+ (geom_line)
           (xlab "x")
           (ylab "y"))))
